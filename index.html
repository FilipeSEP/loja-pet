package com.petshop.controller; import org.springframework.stereotype.Controller; import org.springframework.ui.Model; import org.springframework.web.bind.annotation.GetMapping; import org.springframework.web.bind.annotation.PostMapping; import org.springframework.web.bind.annotation.RequestParam; @Controller public class PetShopController { // Simulação de um banco de dados de produtos // Em um projeto real, isso viria de um banco de dados private List produtos = List.of( new Produto("Ração Premium para Cães", "Alimento completo e balanceado para cães adultos.", "racao-caes.jpg", 120.50), new Produto("Areia Higiênica para Gatos", "Super absorvente e com controle de odor.", "areia-gatos.jpg", 35.90), new Produto("Brinquedo Mordedor", "Ideal para filhotes, ajuda a aliviar a coceira nas gengivas.", "mordedor.jpg", 25.00), new Produto("Shampoo para Cães e Gatos", "Hipoalergênico e com pH neutro.", "shampoo-pet.jpg", 45.00) ); // Classe para representar um produto (Exemplo de um 'POJO' - Plain Old Java Object) public static class Produto { public String nome; public String descricao; public String imagem; public double preco; public Produto(String nome, String descricao, String imagem, double preco) { this.nome = nome; this.descricao = descricao; this.imagem = imagem; this.preco = preco; } } // Mapeia a URL "/" para a página inicial (index.html) @GetMapping("/") public String paginaInicial(Model model) { model.addAttribute("produtos", produtos); return "index"; // Spring Boot busca o arquivo 'index.html' } // Mapeia a URL "/login" para a lógica de login @PostMapping("/login") public String login(@RequestParam String email, @RequestParam String senha, Model model) { // Lógica de autenticação: verificar no banco de dados se o email e a senha estão corretos // Por enquanto, vamos simular que o login é sempre bem-sucedido boolean loginSucesso = "cliente@teste.com".equals(email) && "12345".equals(senha); if (loginSucesso) { model.addAttribute("mensagem", "Login realizado com sucesso! Bem-vindo(a)!"); // Redireciona para a página principal ou outra página return "redirect:/"; } else { model.addAttribute("erroLogin", "Email ou senha incorretos. Tente novamente."); return "index"; // Volta para a página de login com a mensagem de erro } } }
